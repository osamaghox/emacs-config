(defun my/org-inline-images-from-denote-directory ()
  "اعرض الصور inline لكل الملفات في denote-directory التي تبدأ بـ timestamp pattern.
الصور تظهر في سطر بمفردها بعد النص الأصلي."
  (interactive)
  (when (derived-mode-p 'org-mode)
    (let ((denote-dir (file-truename denote-directory))
          (timestamp-regex "\\b[0-9]\\{8\\}T[0-9]\\{6\\}")
          (max-width 300)
          (max-height 200))
      (save-excursion
        (goto-char (point-min))
        ;; أولًا نحذف أي overlays موجودة مسبقًا لمنع التكرار
        (remove-overlays (point-min) (point-max) 'org-image-overlay t)
        
        ;; نبحث عن السطر الذي يحتوي على الـ timestamp (الـ ID)
        (while (re-search-forward timestamp-regex nil t)
          (let* ((ts (match-string 0))
                 (file (car (directory-files denote-dir t (concat "^" ts)))))
            
            (when (and file (file-exists-p file)
                       (string-match-p (image-file-name-regexp) file))
              
              ;; تحديد نهاية السطر الذي يحتوي على الرابط
              (let* ((line-end (line-end-position)) 
                     (img (create-image file nil nil :width max-width :height max-height))
                     
                     ;; ⭐️ التعديل هنا: نستخدم \n\n لضمان فصل الصورة عن السطر الحالي
                     ;; يجب أن نستخدم `\n` في البداية لضمان أن الصورة تبدأ في سطر جديد
                     ;; ويجب أن نستخدمها مرتين لضمان فصل واضح.
                     (img-str (propertize "\n\n" 'display img)) 
                     
                     ;; إنشاء overlay جديد في نهاية السطر
                     (ov (make-overlay line-end line-end))) 
                
                ;; استخدام 'after-string' هو الأنسب لوضع المحتوى بعد النقطة المحددة
                ;; (يمكن أن يعمل 'before-string' أيضاً في هذا الموضع)
                (overlay-put ov 'after-string img-str) 
                
                (overlay-put ov 'org-image-overlay t)
                (overlay-put ov 'modification-hooks
                             (list (lambda (o _f _s) (delete-overlay o))))))))))))

(add-hook 'org-mode-hook #'my/org-inline-images-from-denote-directory)
