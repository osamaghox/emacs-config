* The early initialisation of Emacs (early-init.el)

This is the first file that Emacs reads when starting up. It should contain code that does not depend on any package or the proportions of the Emacs frame. In general, this early initialisation file is meant to set up a few basic things before Emacs produces the initial frame by delegating to the init.el

** The =early-init.el= section to set PATH and system tools

#+begin_src emacs-lisp :tangle "early-init.el" :mkdirp yes
;; Detect system type and add appropriate paths
(let ((path-sep (if (eq system-type 'windows-nt) ";" ":")))

  ;; Android (Termux) settings
  (when (string-equal system-type "android")
    (setenv "PATH" (format "%s%s%s"
                           "/data/data/com.termux/files/usr/bin"
                           path-sep
                           (getenv "PATH")))
    (setenv "LD_LIBRARY_PATH" (format "%s:%s"
                                      "/data/data/com.termux/files/usr/lib"
                                      (getenv "LD_LIBRARY_PATH")))
    (push "/data/data/com.termux/files/usr/bin" exec-path))

  ;; Windows settings
  (when (eq system-type 'windows-nt)
    ;; Git
    (add-to-list 'exec-path "~/sync-thing/online-sync/emacs/emacs-data/tools/git/cmd")
    (setenv "PATH" (concat (expand-file-name "~/sync-thing/online-sync/emacs/emacs-data/tools/git/cmd/")
                           path-sep
                           (getenv "PATH")))

    ;; fd (file search tool)
    (add-to-list 'exec-path "~/sync-thing/online-sync/emacs/emacs-data/tools/fd/")

    ;; ripgrep
    (add-to-list 'exec-path "~/sync-thing/online-sync/emacs/emacs-data/tools/ripgrep/")))
#+end_src

*** The =osama-emacs-essentials.el= section for UI basics

#+begin_src emacs-lisp :tangle "osama-emacs-essentials.el" :mkdirp yes
;; Visible bell
(setq visible-bell t)

;; Show current column number
(column-number-mode)

;; Enable line numbers globally and disable in specific modes
(global-display-line-numbers-mode t)
(dolist (mode '(org-mode-hook
                term-mode-hook
                shell-mode-hook
                eshell-mode-hook))
  (add-hook mode (lambda () (display-line-numbers-mode 0))))

;; Make cursor a vertical bar
(setq-default cursor-type 'bar)

;; Highlight current line
(global-hl-line-mode 1)

*** The =osama-emacs-essentials.el= section for Session, Auto-Save & Backup Management

#+begin_src emacs-lisp :tangle "osama-emacs-essentials.el" :mkdirp yes
;; Define cache directory
(defvar osama-emacs-cache-dir
  (expand-file-name "~/osama-emacs/emacs-cache/")
  "Portable directory for saving Emacs session cache across devices.")

;; Ensure directory exists
(defun osama-ensure-directory (dir)
  (unless (file-directory-p dir)
    (make-directory dir t)))

;; Prepare subdirectories
(let ((dirs '("savehist" "recentf" "desktop" "backups" "autosaves")))
  (dolist (sub dirs)
    (osama-ensure-directory (expand-file-name sub osama-emacs-cache-dir))))

;; Savehist: command history
(setq savehist-file (expand-file-name "savehist/savehist" osama-emacs-cache-dir))
(savehist-mode 1)

;; Recentf: recent files list
(setq recentf-save-file (expand-file-name "recentf/recentf" osama-emacs-cache-dir))
(recentf-mode 1)

;; Desktop: full session restore
(setq desktop-path           (list (expand-file-name "desktop" osama-emacs-cache-dir))
      desktop-dirname        (expand-file-name "desktop" osama-emacs-cache-dir)
      desktop-base-file-name "emacs-desktop")
(desktop-save-mode 1)

;; Backups: store ~ files in backups/
(setq backup-directory-alist
      `((".*" . ,(expand-file-name "backups/" osama-emacs-cache-dir)))
      backup-by-copying t
      version-control t
      delete-old-versions t
      kept-new-versions 5
      kept-old-versions 2)

;; Auto-saves: store #files# in autosaves/
(setq auto-save-file-name-transforms
      `((".*" ,(concat (expand-file-name "autosaves/" osama-emacs-cache-dir) "\\1") t))
      auto-save-default t
      auto-save-timeout 20
      auto-save-interval 200)

;; Optional: disable lockfiles
(setq create-lockfiles nil)
#+end_src











































** The =osama-emacs-packages.el= section for package management

#+begin_src emacs-lisp :tangle "init.el" :mkdirp yes
;; Prevent Emacs from loading packages before init
(setq package-enable-at-startup nil)

(require 'package)

;; Package repositories
(setq package-archives
      '(("melpa" . "https://melpa.org/packages/")
        ("org"   . "https://orgmode.org/elpa/")
        ("elpa"  . "https://elpa.gnu.org/packages/")))

(package-initialize)

;; Install use-package if not installed
(unless (package-installed-p 'use-package)
  (unless package-archive-contents
    (package-refresh-contents))
  (package-install 'use-package))

(require 'use-package)

;; Automatically install declared packages
(setq use-package-always-ensure t)

(provide 'osama-emacs-packages)
#+end_src
